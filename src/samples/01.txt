camera.position.copy(ref.current.position);

      const direction = new Vector3();

      const frontVector = new Vector3(
        0,
        0,
        (moveBackward ? 1 : 0) - (moveForward ? 1 : 0)
      );
      const sideVector = new Vector3(
        (moveLeft ? 1 : 0) - (moveRight ? 1 : 0),
        0,
        0
      );

      direction
        .subVectors(frontVector, sideVector)
        .normalize()
        .multiplyScalar(0.1)
        .applyEuler(camera.rotation);

      if (ref.current.position.z > topClosest) {
        ref.current.position.set(direction.x + ref.current.position.x, 0, direction.z + ref.current.position.z);
      }

      camera?.position.set(ref.current.position.x, 0.5, ref.current.position.z);
